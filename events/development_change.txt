country_event = {
	id = countrydevelopment.1
	title = countrydevelopment.1.t
	desc = countrydevelopment.1.d
	fire_only_once = no
	trigger = {
        check_variable = { 
			var = government_level_var  
			value = 180 
			compare = equals
		}
		NOT = { has_idea = government_level6 }
	}
	option = { 
		name = countrydevelopment.1.a
        
		add_to_variable = { government_level_var = -180 }
		add_to_variable = { government_increase_var = -5 }
		clamp_variable = { 
			var = government_increase_var 
			min = 0
			max = 99
		}
		development_frame_data_settlement = yes
		if = {
			limit = { has_idea = government_level1 }
			swap_ideas = {
				remove_idea = government_level1
				add_idea = government_level2
			}
		}
		else_if = {
			limit = { has_idea = government_level2 }
			swap_ideas = {
				remove_idea = government_level2
				add_idea = government_level3
			}
		}
		else_if = {
			limit = { has_idea = government_level3 }
			swap_ideas = {
				remove_idea = government_level3
				add_idea = government_level4
			}
		}
		else_if = {
			limit = { has_idea = government_level4 }
			swap_ideas = {
				remove_idea = government_level4
				add_idea = government_level5
			}
		}
		else_if = {
			limit = { has_idea = government_level5 }
			swap_ideas = {
				remove_idea = government_level5
				add_idea = government_level6
			}
		}
	}		
}
